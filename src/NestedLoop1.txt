import java.lang.*;
class NestedLoop1
{
                 public static void main(String[] args)
                  {
                       for (int row=1;row<=5;row++)
                        {
                           for (int col=1;col<=6;col++)
                         { 
                            System.out.print(col);
                           }

                        System.out.println();
                       }
          }

}
